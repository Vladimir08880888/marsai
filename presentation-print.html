const decoded = jwt.verify(token, process.env.JWT_SECRET);
const user = await User.findOne({ where: { email: decoded.email } });

if (!user || (roles.length && !roles.includes(user.role))) {

}

API Endpoints

// Authentification
POST  /auth/login           // Connexion (retourne JWT)
POST  /auth/register        // Inscription
POST  /auth/check-token     // Verifier le token

// Public — Page d'accueil
GET   /home/featured        // Films en vedette (selected + finalist)
GET   /home/stats           // Statistiques du festival

// Films
GET   /films                // Liste paginee (page, limit)
POST  /films                // Creer un film

// Admin (protege par AuthMiddleware)
GET   /users                // Gestion des utilisateurs
GET   /overview             // Dashboard stats
POST  /uploads              // Upload fichiers (multer)

Features Cles

🌓

Dark / Light Theme

Responsive Design
